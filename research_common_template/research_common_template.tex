% ═══ Section: Research and Analyze Sources ════════════
\begin{sectionheader}
\sectionfont{\textcolor{white}{Research and Analyze Sources}}
\end{sectionheader}

\vspace{4pt}

\begin{itemize}[leftmargin=1.5em, itemsep=2pt, topsep=4pt]
  \item \textbf{Primary sources:} These are original materials that provide firsthand accounts or direct evidence concerning a topic. In English literature, this includes the novels, poems, plays, and essays written by authors being studied.
  \item \textbf{Secondary sources:} These sources analyze, interpret, or evaluate primary sources. Examples include literary criticism, scholarly articles, and book reviews.
  \item \textbf{Tertiary sources:} These sources compile and summarize information from primary and secondary sources. Examples include encyclopedias, textbooks, and literary guides.
\end{itemize}

\vspace{4pt}

\begin{tipbox}
\textbf{Research Tip:} When doing research, look for primary and secondary sources from different places. Read books, websites, and articles. Ask: Is this information true? Does it make sense? Use what you learn to answer your questions.
\end{tipbox}
